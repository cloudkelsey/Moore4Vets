			ideology = despotism

				#
			}
		}
		GER = { country_event = DOD_yugoslavia.2 }
		if = {
			limit = {
				is_in_faction_with = GER
			}
			GER = { remove_from_faction = YUG }
		}
		if = {
			limit = {
				is_puppet_of = GER
			}
			set_autonomy = {
				target = YUG
				autonomy_state = autonomy_free
			}
			set_politics = { ruling_party = neutrality elections_allowed = yes }

			factor = 30
		}

					limit = {
						104 = { is_owned_by = YUG }
					}
					set_capital = 104
				}
			}
		}
		start_civil_war = { ideology = democratic size = 0.6 capital = 107 states = { 108 107 106 105 } }

			limit = {
				original_tag = YUG
